      import 
      androidx.appcompat.app.AppCompatActivity;
      import android.os.Bundle; 
      import android.view.View;
      import android.widget.Button; 
      import android.widget.TextView; 
      import android.graphics.Color; 
      import android.graphics.Typeface; 
      
      public class MainActivity extends AppCompatActivity { 
        private TextView textCollegeName;
        private Button buttonChangeProperties;
      
        @Override 
        protected void onCreate(Bundle savedInstanceState) { 
          super.onCreate(savedInstanceState); 
          setContentView(R.layout.activity_main);
      
          textCollegeName = findViewById(R.id.textCollegeName); 
          buttonChangeProperties = findViewById(R.id.buttonChangeProperties); 
      
          buttonChangeProperties.setOnClickListener(new View.OnClickListener() { 
            @Override 
            public void onClick(View v) { 
              // Change color to red 
              textCollegeName.setTextColor(Color.RED); 
              // Change font size to 32sp 
              textCollegeName.setTextSize(32);
              // Change font style to italic 
              textCollegeName.setTypeface(null, Typeface.ITALIC);
            } 
          }); 
        } 
      }
